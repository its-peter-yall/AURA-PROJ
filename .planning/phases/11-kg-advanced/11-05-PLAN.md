```python
class GraphOptions(BaseModel):
    include_entity_types: Optional[List[str]] = None
    exclude_entity_types: Optional[List[str]] = None
    include_relationship_types: Optional[List[str]] = None
    exclude_relationship_types: Optional[List[str]] = None
    max_nodes: int = 500
    include_chunks: bool = False
    include_documents: bool = True
    layout: LayoutType = LayoutType.FORCE_DIRECTED
    group_by: Optional[str] = None  # "type", "module", "document"

class LayoutType(str, Enum):
    FORCE_DIRECTED = "force_directed"
    HIERARCHICAL = "hierarchical"
    RADIAL = "radial"
    CIRCULAR = "circular"

class ExportFormat(str, Enum):
    JSON = "json"
    GRAPHML = "graphml"
    GEXF = "gexf"
    CSV = "csv"

class VisualizationNode(BaseModel):
    id: str
    label: str
    type: str
    group: Optional[str] = None
    size: float = 1.0
    color: Optional[str] = None
    x: Optional[float] = None
    y: Optional[float] = None
    properties: Dict[str, Any] = {}

class VisualizationEdge(BaseModel):
    id: str
    source: str
    target: str
    type: str
    weight: float = 1.0
    color: Optional[str] = None
    properties: Dict[str, Any] = {}

class VisualizationGraph(BaseModel):
    nodes: List[VisualizationNode]
    edges: List[VisualizationEdge]
    metadata: GraphMetadata
    layout_applied: bool = False

class GraphMetadata(BaseModel):
    module_ids: List[str]
    node_count: int
    edge_count: int
    entity_type_counts: Dict[str, int]
    relationship_type_counts: Dict[str, int]
    generated_at: datetime
    options_used: GraphOptions
```

Layout algorithms:
- Force-directed: Default, good for general exploration
- Hierarchical: For document → chunk → entity relationships
- Radial: For entity neighborhood exploration
- Circular: For module comparison

```typescript
import React, { useRef, useState, useCallback, useMemo } from 'react';
import ForceGraph2D from 'react-force-graph-2d';
import { VisualizationGraph, VisualizationNode, GraphOptions } from '../types/kg-query.types';

interface UnifiedGraphViewProps {
  graph: VisualizationGraph;
  onNodeClick?: (node: VisualizationNode) => void;
  onNodeHover?: (node: VisualizationNode | null) => void;
  options?: GraphViewOptions;
  height?: number;
  width?: number;
}

interface GraphViewOptions {
  showLabels: boolean;
  labelSize: number;
  nodeSize: 'fixed' | 'by_connections' | 'by_property';
  nodeSizeProperty?: string;
  edgeWidth: 'fixed' | 'by_weight';
  enableZoom: boolean;
  enablePan: boolean;
  highlightOnHover: boolean;
}

const ENTITY_COLORS: Record<string, string> = {
  Topic: '#4CAF50',
  Concept: '#2196F3',
  Methodology: '#FF9800',
  Finding: '#9C27B0',
  Definition: '#00BCD4',
  Document: '#795548',
  Chunk: '#607D8B',
  Module: '#3F51B5',
};

export function UnifiedGraphView({
  graph,
  onNodeClick,
  onNodeHover,
  options = defaultOptions,
  height = 600,
  width,
}: UnifiedGraphViewProps) {
  const graphRef = useRef();
  const [selectedNode, setSelectedNode] = useState<VisualizationNode | null>(null);
  const [highlightNodes, setHighlightNodes] = useState(new Set());
  const [highlightLinks, setHighlightLinks] = useState(new Set());

  // Transform data for react-force-graph
  const graphData = useMemo(() => ({
    nodes: graph.nodes.map(n => ({
      ...n,
      color: ENTITY_COLORS[n.type] || '#999',
    })),
    links: graph.edges.map(e => ({
      source: e.source,
      target: e.target,
      type: e.type,
      ...e,
    })),
  }), [graph]);

  // Node rendering
  const nodeCanvasObject = useCallback((node, ctx, globalScale) => {
    // Custom node rendering with label
  }, [options]);

  // Highlight on hover
  const handleNodeHover = useCallback((node) => {
    // Set highlight state
    onNodeHover?.(node);
  }, [onNodeHover]);

  return (
    <div className="unified-graph-view">
      <GraphControls 
        onZoomIn={() => graphRef.current?.zoom(1.5)}
        onZoomOut={() => graphRef.current?.zoom(0.67)}
        onCenter={() => graphRef.current?.centerAt(0, 0)}
      />
      <ForceGraph2D
        ref={graphRef}
        graphData={graphData}
        height={height}
        width={width}
        nodeCanvasObject={nodeCanvasObject}
        onNodeClick={onNodeClick}
        onNodeHover={handleNodeHover}
        // ... additional props
      />
      {selectedNode && (
        <NodeDetailsPanel node={selectedNode} onClose={() => setSelectedNode(null)} />
      )}
    </div>
  );
}
```
