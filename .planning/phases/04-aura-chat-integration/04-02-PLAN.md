# routers/student_modules.py
from fastapi import APIRouter, HTTPException, Depends
from pydantic import BaseModel
from typing import Optional, List
from datetime import datetime
from typing import Literal
import httpx
import json

from backend.rag_engine import RAGEngine
from backend.graph_manager import GraphManager

router = APIRouter(prefix="/api/student", tags=["Student Modules"])

# Initialize RAG engine and graph manager
graph_manager = GraphManager()
rag_engine = RAGEngine(graph_manager)

# Redis key prefix for module selections
MODULE_SELECTION_PREFIX = "student:modules:"


class ModuleResponse(BaseModel):
    id: str
    code: str
    name: str
    description: Optional[str] = None
    year: int
    semester: int
    document_count: int
    published_at: Optional[datetime] = None


class AvailableModulesResponse(BaseModel):
    modules: List[ModuleResponse]
    total: int


class SelectModulesRequest(BaseModel):
    module_ids: List[str]
    session_id: str = "default"


class SelectModulesResponse(BaseModel):
    selected: List[str]
    count: int


class StudentQueryRequest(BaseModel):
    query: str
    module_ids: Optional[List[str]] = None
    session_id: str = "default"
    mode: Literal["assistant", "tutor"] = "assistant"
    enable_cross_module: bool = True
    enable_thinking: bool = False


class QueryResponse(BaseModel):
    answer: str
    sources: List[dict]
    cross_concepts: List[dict]
    modules_queried: List[str]
    model_used: str
    thought_summary: Optional[str] = None


def get_redis():
    """Get Redis client for session storage."""
    import redis
    return redis.Redis(host="127.0.0.1", port=6379, db=0, decode_responses=True)


async def get_selected_modules(session_id: str) -> List[str]:
    """Get selected modules for a session from Redis."""
    try:
        r = get_redis()
        key = f"{MODULE_SELECTION_PREFIX}{session_id}"
        data = r.get(key)
        if data:
            return json.loads(data)
        return []
    except Exception:
        return []


async def set_selected_modules(session_id: str, module_ids: List[str]) -> None:
    """Store selected modules for a session in Redis (24h TTL)."""
    try:
        r = get_redis()
        key = f"{MODULE_SELECTION_PREFIX}{session_id}"
        r.setex(key, 86400, json.dumps(module_ids))
    except Exception as e:
        print(f"Failed to store module selection: {e}")


@router.get("/modules/available", response_model=AvailableModulesResponse)
async def get_available_modules():
    """
    Fetch published modules from AURA-NOTES-MANAGER API.
    """
    try:
        async with httpx.AsyncClient() as client:
            response = await client.get(
                "http://127.0.0.1:8001/api/v1/modules/published",
                timeout=10.0
            )
            if response.status_code != 200:
                return {"modules": [], "total": 0}

            data = response.json()
            modules = data.get("modules", [])

            return {
                "modules": [
                    {
                        "id": m.get("id", ""),
                        "code": m.get("code", ""),
                        "name": m.get("name", ""),
                        "description": m.get("description"),
                        "year": m.get("year", 2024),
                        "semester": m.get("semester", 1),
                        "document_count": m.get("document_count", 0),
                        "published_at": m.get("published_at"),
                    }
                    for m in modules
                ],
                "total": len(modules)
            }
    except Exception as e:
        print(f"Failed to fetch available modules: {e}")
        return {"modules": [], "total": 0}


@router.get("/modules/selected", response_model=SelectModulesResponse)
async def get_selected_modules_endpoint(session_id: str = "default"):
    """
    Get currently selected modules for a study session.
    """
    module_ids = await get_selected_modules(session_id)
    return {"selected": module_ids, "count": len(module_ids)}


@router.post("/modules/select", response_model=SelectModulesResponse)
async def select_modules(request: SelectModulesRequest):
    """
    Select modules for study. Replaces current selection.
    """
    await set_selected_modules(request.session_id, request.module_ids)
    return {"selected": request.module_ids, "count": len(request.module_ids)}


@router.post("/modules/deselect", response_model=SelectModulesResponse)
async def deselect_modules(
    module_ids: List[str],
    session_id: str = "default"
):
    """
    Deselect specific modules from current selection.
    """
    current = await get_selected_modules(session_id)
    remaining = [m for m in current if m not in module_ids]
    await set_selected_modules(session_id, remaining)
    return {"selected": remaining, "count": len(remaining)}


@router.post("/modules/clear", response_model=SelectModulesResponse)
async def clear_modules(session_id: str = "default"):
    """
    Clear all selected modules.
    """
    await set_selected_modules(session_id, [])
    return {"selected": [], "count": 0}


@router.post("/query", response_model=QueryResponse)
async def student_query(request: StudentQueryRequest):
    """
    Execute RAG query with module filtering.
    Uses selected modules if module_ids not specified.
    """
    # Get selected modules if not specified
    module_ids = request.module_ids
    if not module_ids:
        module_ids = await get_selected_modules(request.session_id)

    if not module_ids:
        raise HTTPException(
            status_code=400,
            detail="No modules selected. Use /modules/select first or specify module_ids."
        )

    # Execute module-aware RAG query
    result = rag_engine.query(
        user_query=request.query,
        session_id=request.session_id,
        mode=request.mode,
        module_ids=module_ids,
        enable_cross_module=request.enable_cross_module,
        enable_thinking=request.enable_thinking,
    )

    return {
        "answer": result.get("answer", ""),
        "sources": result.get("sources", []),
        "cross_concepts": result.get("cross_concepts", []),
        "modules_queried": module_ids,
        "model_used": result.get("model_used", "gemini-2.0-flash-001"),
        "thought_summary": result.get("thought_summary"),
    }
