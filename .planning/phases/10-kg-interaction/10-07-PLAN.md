import { useMutation, useQuery } from '@tanstack/react-query';
import { SearchRequest, SearchResponse, GraphData } from '../types/kg-query.types';
import { kgQueryApi } from '../api/kg-query.api';

export function useKGQuery() {
  const searchMutation = useMutation({
    mutationFn: (request: SearchRequest) => kgQueryApi.search(request),
    onError: (error) => {
      console.error('Search failed:', error);
    }
  });

  const multiDocMutation = useMutation({
    mutationFn: (request: MultiDocRequest) => kgQueryApi.multiDocQuery(request),
  });

  return {
    search: searchMutation.mutate,
    searchAsync: searchMutation.mutateAsync,
    isSearching: searchMutation.isPending,
    searchResults: searchMutation.data,
    searchError: searchMutation.error,
    
    multiDocQuery: multiDocMutation.mutate,
    isMultiDocQuerying: multiDocMutation.isPending,
    multiDocResults: multiDocMutation.data,
  };
}

export function useGraphData(moduleId?: string, entityTypes?: string[]) {
  return useQuery({
    queryKey: ['graph-data', moduleId, entityTypes],
    queryFn: () => kgQueryApi.getGraphData(moduleId, entityTypes),
    enabled: !!moduleId,
    staleTime: 5 * 60 * 1000, // 5 minutes
  });
}

export function useGraphSchema() {
  return useQuery({
    queryKey: ['graph-schema'],
    queryFn: () => kgQueryApi.getGraphSchema(),
    staleTime: 30 * 60 * 1000, // 30 minutes
  });
}

Create API client for KG query endpoints.

Purpose: HTTP client functions for communicating with backend KG endpoints.

```typescript
import { api } from '@/lib/api';
import { 
  SearchRequest, 
  SearchResponse, 
  GraphData, 
  GraphSchema,
  ResultFeedback,
  AnswerFeedback 
} from '../types/kg-query.types';

const BASE_PATH = '/v1/kg';

export const kgQueryApi = {
  search: async (request: SearchRequest): Promise<SearchResponse> => {
    const response = await api.post(`${BASE_PATH}/query`, request);
    return response.data;
  },

  multiDocQuery: async (request: MultiDocRequest): Promise<MultiDocResponse> => {
    const response = await api.post(`${BASE_PATH}/query/multi-doc`, request);
    return response.data;
  },

  analyze: async (request: AnalysisRequest): Promise<AnalysisResponse> => {
    const response = await api.post(`${BASE_PATH}/analyze`, request);
    return response.data;
  },

  getGraphSchema: async (): Promise<GraphSchema> => {
    const response = await api.get(`${BASE_PATH}/graph/schema`);
    return response.data;
  },

  getGraphData: async (
    moduleId?: string, 
    entityTypes?: string[], 
    limit: number = 100
  ): Promise<GraphData> => {
    const params = new URLSearchParams();
    if (moduleId) params.append('module_id', moduleId);
    if (entityTypes) entityTypes.forEach(t => params.append('entity_types', t));
    params.append('limit', limit.toString());
    
    const response = await api.get(`${BASE_PATH}/graph/data?${params}`);
    return response.data;
  },

  submitFeedback: async (feedback: ResultFeedback | AnswerFeedback): Promise<void> => {
    await api.post(`${BASE_PATH}/feedback`, feedback);
  },
};
```
