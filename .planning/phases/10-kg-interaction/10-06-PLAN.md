```python
class FeedbackType(str, Enum):
    RESULT_RELEVANCE = "result_relevance"
    ANSWER_QUALITY = "answer_quality"
    ANSWER_ACCURACY = "answer_accuracy"
    CLICK = "click"
    DWELL_TIME = "dwell_time"

class ResultFeedback(BaseModel):
    query: str
    query_hash: str
    result_id: str  # chunk_id or entity_id
    result_rank: int
    relevance_score: float = Field(ge=0.0, le=1.0)  # 0=irrelevant, 1=highly relevant
    user_id: Optional[str] = None
    session_id: Optional[str] = None
    module_ids: List[str] = []
    timestamp: datetime = Field(default_factory=datetime.utcnow)
    feedback_type: FeedbackType = FeedbackType.RESULT_RELEVANCE

class AnswerFeedback(BaseModel):
    query: str
    query_hash: str
    answer_hash: str
    helpful: bool
    accuracy_score: Optional[float] = None  # 0=inaccurate, 1=accurate
    user_id: Optional[str] = None
    session_id: Optional[str] = None
    module_ids: List[str] = []
    comment: Optional[str] = None
    timestamp: datetime = Field(default_factory=datetime.utcnow)
    feedback_type: FeedbackType = FeedbackType.ANSWER_QUALITY

class FeedbackStats(BaseModel):
    total_feedback_count: int
    positive_feedback_ratio: float
    average_relevance_score: float
    feedback_by_type: Dict[str, int]
    feedback_by_module: Dict[str, int]
    time_range: Optional[tuple] = None
```

Neo4j storage approach:
Create Feedback nodes linked to queries and results:
```cypher
CREATE (f:Feedback {
    id: $feedback_id,
    query_hash: $query_hash,
    feedback_type: $feedback_type,
    score: $score,
    user_id: $user_id,
    timestamp: datetime()
})
WITH f
MATCH (r) WHERE r.id = $result_id
MERGE (f)-[:FEEDBACK_FOR]->(r)
```

Query hash for grouping similar queries:
- Normalize query (lowercase, strip punctuation)
- Compute SHA-256 hash
- Use hash for grouping feedback on similar queries
