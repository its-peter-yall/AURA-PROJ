    **File 2: AURA-CHAT/client/src/features/settings/api/settingsApi.ts** (NEW)
    Create with file header. Uses `api` from `@/lib/api` (the existing axios instance). Typed fetch wrappers for all settings endpoints:
    ```typescript
    import api from '@/lib/api';
    import type { ModelInfo, DefaultModelSetting, ApiKeyStatus } from '@/types/settings';

    export const fetchModels = async (): Promise<ModelInfo[]> => {
        const response = await api.get<ModelInfo[]>('/api/v1/settings/models');
        return response.data;
    };

    export const fetchDefaults = async (): Promise<Record<string, DefaultModelSetting>> => {
        const response = await api.get<Record<string, DefaultModelSetting>>('/api/v1/settings/defaults');
        return response.data;
    };

    export const updateDefault = async (useCase: string, setting: DefaultModelSetting) => {
        const response = await api.put(`/api/v1/settings/defaults/${useCase}`, setting);
        return response.data;
    };

    export const fetchApiKeyStatus = async (provider: string): Promise<ApiKeyStatus> => {
        const response = await api.get<ApiKeyStatus>(`/api/v1/settings/providers/${provider}/api-key`);
        return response.data;
    };

    export const storeApiKey = async (provider: string, api_key: string): Promise<ApiKeyStatus> => {
        const response = await api.post<ApiKeyStatus>(`/api/v1/settings/providers/${provider}/api-key`, { api_key });
        return response.data;
    };

    export const deleteApiKey = async (provider: string): Promise<{ status: string }> => {
        const response = await api.delete(`/api/v1/settings/providers/${provider}/api-key`);
        return response.data;
    };

    export const validateApiKey = async (provider: string) => {
        const response = await api.post(`/api/v1/settings/providers/${provider}/validate`);
        return response.data as { provider: string; valid: boolean; error?: string };
    };
    ```

    ```typescript
    import { useAuthStore } from '@/stores/useAuthStore';

    export function AdminSettingsRoute() {
        const { isAuthenticated, isAdmin, isInitialized } = useAuthStore();

        if (!isInitialized) {
            return (
                <div className="min-h-dvh flex items-center justify-center bg-[#0a0a0a]">
                    <div className="flex flex-col items-center gap-4">
                        <div className="w-10 h-10 border-2 border-cyber-yellow/30 border-t-cyber-yellow rounded-full animate-spin" />
                        <p className="text-sm text-neutral-500">Loading...</p>
                    </div>
                </div>
            );
        }

        if (!isAuthenticated()) {
            return <Navigate to="/login" replace />;
        }

        if (!isAdmin()) {
            return <Navigate to="/" replace />;
        }

        return <Outlet />;
    }
    ```
