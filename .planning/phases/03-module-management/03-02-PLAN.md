# modules/publishing.py
from typing import Dict, Any, Optional
from datetime import datetime
from google.cloud import firestore
from .service import ModuleService
from .models import ModuleStatus

class ModulePublisher:
    """Service for module publishing workflow."""

    PUBLISHED_MODULES_COLLECTION = "published_modules"

    def __init__(self, db: firestore.Client = None):
        self.db = db or firestore.Client()
        self.module_service = ModuleService(db)
        self.published_collection = self.db.collection(self.PUBLISHED_MODULES_COLLECTION)
        self.audit_collection = self.db.collection("module_audit_log")

    async def publish(self, module_id: str, staff_id: str) -> Dict[str, Any]:
        """
        Publish a module for student access.

        Workflow:
        1. Validate module exists and is in DRAFT status
        2. Check module has at least 1 document
        3. Update status to PUBLISHED
        4. Add to published_modules collection (for AURA-CHAT)
        5. Log audit trail
        6. Trigger document processing for any pending documents

        Args:
            module_id: Module to publish
            staff_id: Staff member publishing the module

        Returns:
            Published module data
        """
        # Get module
        module = await self.module_service.get_by_id(module_id)
        if not module:
            raise ValueError(f"Module {module_id} not found")

        # Validate status
        if module.get("status") != ModuleStatus.DRAFT.value:
            raise ValueError(f"Module must be in DRAFT status to publish")

        # Check document count
        if module.get("document_count", 0) < 1:
            raise ValueError("Module must have at least 1 document to publish")

        now = datetime.utcnow()

        # Update status
        await self.module_service.update(
            module_id,
            {"status": ModuleStatus.PUBLISHED}
        )

        # Add to published modules (for AURA-CHAT)
        published_doc = {
            "module_id": module_id,
            "name": module["name"],
            "code": module["code"],
            "description": module.get("description"),
            "year": module["year"],
            "semester": module["semester"],
            "document_count": module.get("document_count", 0),
            "published_at": now,
            "published_by": staff_id,
            "student_access": True
        }
        self.published_collection.document(module_id).set(published_doc)

        # Audit log
        self.audit_collection.add({
            "module_id": module_id,
            "action": "PUBLISH",
            "performed_by": staff_id,
            "timestamp": now,
            "details": {
                "document_count": module.get("document_count", 0),
                "previous_status": ModuleStatus.DRAFT.value,
                "new_status": ModuleStatus.PUBLISHED.value
            }
        })

        return {
            **module,
            "status": ModuleStatus.PUBLISHED.value,
            "published_at": now.isoformat()
        }

    async def unpublish(self, module_id: str, staff_id: str, reason: str = None) -> Dict[str, Any]:
        """
        Unpublish a module (hide from students).

        Args:
            module_id: Module to unpublish
            staff_id: Staff member unpublishing
            reason: Reason for unpublishing
        """
        module = await self.module_service.get_by_id(module_id)
        if not module:
            raise ValueError(f"Module {module_id} not found")

        if module.get("status") != ModuleStatus.PUBLISHED.value:
            raise ValueError("Module must be PUBLISHED to unpublish")

        now = datetime.utcnow()

        # Update status
        await self.module_service.update(
            module_id,
            {"status": ModuleStatus.DRAFT.value}
        )

        # Remove from published modules (AURA-CHAT won't see it)
        self.published_collection.document(module_id).delete()

        # Audit log
        self.audit_collection.add({
            "module_id": module_id,
            "action": "UNPUBLISH",
            "performed_by": staff_id,
            "timestamp": now,
            "details": {
                "reason": reason,
                "previous_status": ModuleStatus.PUBLISHED.value,
                "new_status": ModuleStatus.DRAFT.value
            }
        })

        return {
            **module,
            "status": ModuleStatus.DRAFT.value,
            "unpublished_at": now.isoformat()
        }

    async def get_published_modules(self) -> list:
        """Get all published modules for AURA-CHAT."""
        docs = self.published_collection.where("student_access", "==", True).get()
        return [d.to_dict() for d in docs]

    async def get_audit_log(self, module_id: str, limit: int = 50) -> list:
        """Get audit log for a module."""
        docs = (
            self.audit_collection
            .where("module_id", "==", module_id)
            .order_by("timestamp", direction=firestore.Query.DESCENDING)
            .limit(limit)
            .get()
        )
        return [d.to_dict() for d in docs]

from .publishing import ModulePublisher

# Add publisher dependency
def get_module_publisher() -> ModulePublisher:
    return ModulePublisher()

@router.post("/{module_id}/publish", response_model=ModuleResponse)
async def publish_module(
    module_id: str,
    staff_id: str = "staff_user_001",  # TODO: Get from auth
    service: ModuleService = Depends(get_module_service),
    publisher: ModulePublisher = Depends(get_module_publisher)
):
    """Publish a module for student access."""
    try:
        return await publisher.publish(module_id, staff_id)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))

@router.post("/{module_id}/unpublish", response_model=ModuleResponse)
async def unpublish_module(
    module_id: str,
    reason: Optional[str] = None,
    staff_id: str = "staff_user_001",
    service: ModuleService = Depends(get_module_service),
    publisher: ModulePublisher = Depends(get_module_publisher)
):
    """Unpublish a module (hide from students)."""
    try:
        return await publisher.unpublish(module_id, staff_id, reason)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))

@router.get("/{module_id}/audit-log")
async def get_module_audit_log(
    module_id: str,
    limit: int = 50,
    publisher: ModulePublisher = Depends(get_module_publisher)
):
    """Get audit log for a module."""
    return await publisher.get_audit_log(module_id, limit)
