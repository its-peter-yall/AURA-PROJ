# Redis connection settings
REDIS_HOST: str = os.environ.get('REDIS_HOST', '127.0.0.1')

MODULE_LIST_CACHE_TTL: int = 300       # 5 minutes for lists
PUBLISHED_MODULES_TTL: int = 300       # 5 minutes for published list

    if REDIS_PASSWORD:
        return f"redis://:{REDIS_PASSWORD}@{REDIS_HOST}:{REDIS_PORT}/{REDIS_DB}"
    return f"redis://{REDIS_HOST}:{REDIS_PORT}/{REDIS_DB}"

# cache/client.py
import redis
from typing import Optional
from .config import get_redis_url, REDIS_HOST, REDIS_PORT, REDIS_DB

class RedisClient:
    """Redis client wrapper with connection management."""

    _instance: Optional['RedisClient'] = None
    _client: Optional[redis.Redis] = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance

    def __init__(self):
        if self._client is None:
            self._client = redis.Redis(
                host=REDIS_HOST,
                port=REDIS_PORT,
                db=REDIS_DB,
                decode_responses=True,
                socket_connect_timeout=5,
                socket_timeout=5
            )

    @property
    def client(self) -> redis.Redis:
        """Get the Redis client instance."""
        return self._client

    def ping(self) -> bool:
        """Check Redis connection."""
        try:
            return self._client.ping()
        except redis.ConnectionError:
            return False

    def get(self, key: str) -> Optional[str]:
        """Get value from Redis."""
        try:
            return self._client.get(key)
        except redis.ConnectionError:
            return None

    def set(self, key: str, value: str, ttl: int = None) -> bool:
        """Set value in Redis with optional TTL."""
        try:
            if ttl:
                return self._client.setex(key, ttl, value)
            return self._client.set(key, value)
        except redis.ConnectionError:
            return False

    def delete(self, *keys: str) -> int:
        """Delete keys from Redis."""
        try:
            return self._client.delete(*keys)
        except redis.ConnectionError:
            return 0

    def exists(self, *keys: str) -> int:
        """Check if keys exist."""
        try:
            return self._client.exists(*keys)
        except redis.ConnectionError:
            return 0

    def flush_pattern(self, pattern: str) -> int:
        """Delete all keys matching pattern."""
        try:
            keys = self._client.keys(pattern)
            if keys:
                return self._client.delete(*keys)
            return 0
        except redis.ConnectionError:
            return 0

# Global instances
redis_client = RedisClient()
redis_cache = redis_client.client
```

# cache/module_cache.py
import json
import hashlib
from typing import Optional, Dict, Any, List
from .client import RedisClient
from .config import MODULE_CACHE_TTL, MODULE_LIST_CACHE_TTL, PUBLISHED_MODULES_TTL

class ModuleCache:
    """Caching service for module data."""

    _instance: Optional['ModuleCache'] = None
    _redis: RedisClient = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._redis = RedisClient()
        return cls._instance

    def _make_key(self, key_type: str, *args) -> str:
        """Generate cache key."""
        parts = [key_type] + [str(a) for a in args]
        return ":".join(parts)

    def _hash_filters(self, filters: Dict) -> str:
        """Create hash of filter parameters."""
        sorted_items = sorted(filters.items())
        filter_str = "&".join(f"{k}={v}" for k, v in sorted_items)
        return hashlib.md5(filter_str.encode()).hexdigest()[:8]

    # Module single cache
    def get_module(self, module_id: str) -> Optional[Dict[str, Any]]:
        """Get cached module by ID."""
        key = self._make_key("module", module_id)
        data = self._redis.get(key)
        return json.loads(data) if data else None

    def set_module(self, module_id: str, data: Dict[str, Any], ttl: int = None) -> bool:
        """Cache module data."""
        key = self._make_key("module", module_id)
        ttl = ttl or MODULE_CACHE_TTL
        return self._redis.set(key, json.dumps(data), ttl)

    def invalidate_module(self, module_id: str) -> int:
        """Invalidate cached module."""
        key = self._make_key("module", module_id)
        return self._redis.delete(key)

    # Module list cache
    def get_module_list(self, filters: Dict = None) -> Optional[Dict[str, Any]]:
        """Get cached module list."""
        filter_hash = self._hash_filters(filters or {})
        key = self._make_key("modules:list", filter_hash)
        data = self._redis.get(key)
        return json.loads(data) if data else None

    def set_module_list(self, filters: Dict, data: Dict[str, Any], ttl: int = None) -> bool:
        """Cache module list."""
        filter_hash = self._hash_filters(filters)
        key = self._make_key("modules:list", filter_hash)
        ttl = ttl or MODULE_LIST_CACHE_TTL
        return self._redis.set(key, json.dumps(data), ttl)

    def invalidate_module_lists(self) -> int:
        """Invalidate all module list caches."""
        return self._redis.flush_pattern("modules:list:*")

    # Published modules cache (for AURA-CHAT)
    def get_published_modules(self) -> Optional[List[Dict[str, Any]]]:
        """Get cached published modules."""
        key = self._make_key("published", "modules")
        data = self._redis.get(key)
        return json.loads(data) if data else None

    def set_published_modules(self, modules: List[Dict[str, Any]], ttl: int = None) -> bool:
        """Cache published modules list."""
        key = self._make_key("published", "modules")
        ttl = ttl or PUBLISHED_MODULES_TTL
        return self._redis.set(key, json.dumps(modules), ttl)

    def invalidate_published_modules(self) -> int:
        """Invalidate published modules cache."""
        key = self._make_key("published", "modules")
        return self._redis.delete(key)

    # Global invalidation
    def invalidate_all(self, module_id: str = None) -> Dict[str, int]:
        """Invalidate all caches for a module."""
        results = {}
        if module_id:
            results["module"] = self.invalidate_module(module_id)
        results["module_lists"] = self.invalidate_module_lists()
        results["published_modules"] = self.invalidate_published_modules()
        return results

# Global instance
module_cache = ModuleCache()

def get_module_cache() -> ModuleCache:
    """Get module cache instance."""
    return module_cache
```
