```python
# modules/models.py
from pydantic import BaseModel, Field
from typing import Optional, List
from datetime import datetime
from enum import Enum

class ModuleStatus(str, Enum):
    DRAFT = "draft"
    PUBLISHED = "published"
    ARCHIVED = "archived"

class ModuleCreate(BaseModel):
    """Request model for creating a module."""
    name: str = Field(..., min_length=1, max_length=200)
    code: str = Field(..., min_length=1, max_length=50)
    description: Optional[str] = None
    year: int = Field(..., ge=2000, le=2100)
    semester: int = Field(..., ge=1, le=4)

class ModuleUpdate(BaseModel):
    """Request model for updating a module."""
    name: Optional[str] = Field(None, min_length=1, max_length=200)
    description: Optional[str] = None
    status: Optional[ModuleStatus] = None

class ModuleResponse(BaseModel):
    """Response model for module data."""
    id: str
    name: str
    code: str
    description: Optional[str]
    year: int
    semester: int
    status: ModuleStatus
    document_count: int
    created_at: datetime
    updated_at: datetime

class ModuleListResponse(BaseModel):
    """Response for listing modules."""
    modules: List[ModuleResponse]
    total: int
    page: int
    page_size: int
```

```python
# modules/service.py
from typing import List, Optional, Dict, Any
from datetime import datetime
from google.cloud import firestore
from .models import ModuleCreate, ModuleUpdate, ModuleStatus

class ModuleService:
    """Service for module CRUD operations."""

    COLLECTION = "modules"

    def __init__(self, db: firestore.Client = None):
        self.db = db or firestore.Client()
        self.collection = self.db.collection(self.COLLECTION)

    async def create(self, user_id: str, module_data: ModuleCreate) -> Dict[str, Any]:
        """Create a new module."""
        module_id = f"{module_data.code}_{module_data.year}_S{module_data.semester}"

        doc_ref = self.collection.document(module_id)
        module_doc = {
            "id": module_id,
            "name": module_data.name,
            "code": module_data.code.upper(),
            "description": module_data.description,
            "year": module_data.year,
            "semester": module_data.semester,
            "status": ModuleStatus.DRAFT.value,
            "document_count": 0,
            "created_by": user_id,
            "created_at": datetime.utcnow(),
            "updated_at": datetime.utcnow()
        }

        doc_ref.set(module_doc)
        return module_doc

    async def get_by_id(self, module_id: str) -> Optional[Dict[str, Any]]:
        """Get module by ID."""
        doc = self.collection.document(module_id).get()
        return doc.to_dict() if doc.exists else None

    async def list(
        self,
        user_id: Optional[str] = None,
        status: Optional[ModuleStatus] = None,
        year: Optional[int] = None,
        page: int = 1,
        page_size: int = 20
    ) -> Dict[str, Any]:
        """List modules with filters and pagination."""
        query = self.collection

        if user_id:
            query = query.where("created_by", "==", user_id)
        if status:
            query = query.where("status", "==", status.value)
        if year:
            query = query.where("year", "==", year)

        query = query.order_by("created_at", direction=firestore.Query.DESCENDING)

        total = len(query.get())
        offset = (page - 1) * page_size
        docs = list(query.offset(offset).limit(page_size).get())

        return {
            "modules": [d.to_dict() for d in docs],
            "total": total,
            "page": page,
            "page_size": page_size
        }

    async def update(self, module_id: str, update_data: ModuleUpdate) -> Optional[Dict[str, Any]]:
        """Update module."""
        doc_ref = self.collection.document(module_id)
        doc = doc_ref.get()

        if not doc.exists:
            return None

        update_fields = {k: v for k, v in update_data.model_dump().items() if v is not None}
        update_fields["updated_at"] = datetime.utcnow()

        doc_ref.update(update_fields)
        return doc_ref.get().to_dict()

    async def delete(self, module_id: str) -> bool:
        """Delete module (soft delete by archiving)."""
        doc_ref = self.collection.document(module_id)
        doc = doc_ref.get()

        if not doc.exists:
            return False

        doc_ref.update({
            "status": ModuleStatus.ARCHIVED.value,
            "updated_at": datetime.utcnow()
        })
        return True

    async def increment_document_count(self, module_id: str, delta: int = 1) -> bool:
        """Increment/decrement document count for module."""
        doc_ref = self.collection.document(module_id)
        doc = doc_ref.get()

        if not doc.exists:
            return False

        current_count = doc.to_dict().get("document_count", 0)
        doc_ref.update({
            "document_count": current_count + delta,
            "updated_at": datetime.utcnow()
        })
        return True
```

```python
# modules/router.py
from fastapi import APIRouter, Depends, HTTPException, status
from typing import Optional
from .service import ModuleService
from .models import (
    ModuleCreate, ModuleUpdate, ModuleResponse,
    ModuleListResponse, ModuleStatus
)

router = APIRouter(prefix="/modules", tags=["Modules"])

def get_module_service() -> ModuleService:
    """Dependency for getting ModuleService instance."""
    return ModuleService()

@router.post("", response_model=ModuleResponse, status_code=status.HTTP_201_CREATED)
async def create_module(
    module_data: ModuleCreate,
    user_id: str = "staff_user_001",  # TODO: Get from auth
    service: ModuleService = Depends(get_module_service)
):
    """Create a new module."""
    try:
        module = await service.create(user_id, module_data)
        return module
    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

@router.get("", response_model=ModuleListResponse)
async def list_modules(
    status: Optional[ModuleStatus] = None,
    year: Optional[int] = None,
    page: int = 1,
    page_size: int = 20,
    service: ModuleService = Depends(get_module_service)
):
    """List modules with optional filters."""
    return await service.list(status=status, year=year, page=page, page_size=page_size)

@router.get("/{module_id}", response_model=ModuleResponse)
async def get_module(
    module_id: str,
    service: ModuleService = Depends(get_module_service)
):
    """Get a module by ID."""
    module = await service.get_by_id(module_id)
    if not module:
        raise HTTPException(status_code=404, detail="Module not found")
    return module

@router.put("/{module_id}", response_model=ModuleResponse)
async def update_module(
    module_id: str,
    update_data: ModuleUpdate,
    service: ModuleService = Depends(get_module_service)
):
    """Update a module."""
    module = await service.update(module_id, update_data)
    if not module:
        raise HTTPException(status_code=404, detail="Module not found")
    return module

@router.delete("/{module_id}", status_code=status.HTTP_204_NO_CONTENT)
async def delete_module(
    module_id: str,
    service: ModuleService = Depends(get_module_service)
):
    """Delete (archive) a module."""
    success = await service.delete(module_id)
    if not success:
        raise HTTPException(status_code=404, detail="Module not found")
```
