```python
@app.task(
    bind=True,
    autoretry_for=(ConnectionError, TimeoutError),
    retry_backoff=True,
    retry_backoff_max=600,
    retry_jitter=True,
    max_retries=5,
    acks_late=True,
    reject_on_worker_lost=True,
    time_limit=1800,
    soft_time_limit=1500
)
def process_document_task(self, document_id: str, module_id: str, user_id: str):
    """
    Process single document with status tracking.

    Steps with progress:
    - PENDING: 0%
    - PROCESSING: 5% (start)
    - PARSING: 10%
    - CHUNKING: 30%
    - EMBEDDING: 50%
    - EXTRACTING: 70%
    - STORING: 90%
    - READY: 100%
    - FAILED: error state
    """
    try:
        # Check if already processed (idempotency)
        doc = db.collection("documents").document(document_id).get()
        if doc.exists and doc.to_dict().get("kg_status") == "ready":
            return {"document_id": document_id, "status": "skipped", "reason": "already_processed"}

        # Update status to PROCESSING
        self.update_document_status(document_id, "processing", progress=5, step="starting")

        # Process document
        result = await processor.process_document(document_id, module_id, user_id)

        # Update status to READY
        self.update_document_status(
            document_id,
            "ready",
            progress=100,
            step="complete",
            chunk_count=result.get("chunk_count"),
            entity_count=result.get("entity_count")
        )

        return {"document_id": document_id, "status": "completed"}

    except Exception as e:
        self.update_document_status(document_id, "failed", error=str(e))
        raise
```

```python
# kg/router.py
from fastapi import APIRouter, Depends, HTTPException, status
from typing import List, Optional
from datetime import datetime
from google.cloud import firestore

from api.modules.models import (
    DocumentKGStatus, KGStatus,
    BatchProcessingRequest, BatchProcessingResponse,
    ProcessingQueueItem
)
from api.tasks.document_processing_tasks import process_batch_task

router = APIRouter(prefix="/kg", tags=["KG Processing"])

db = firestore.Client()

@router.get("/documents/{document_id}/status", response_model=DocumentKGStatus)
async def get_document_kg_status(document_id: str):
    """
    Get KG status for a single document.

    Returns:
    - pending: Document not yet processed
    - processing: Document currently being processed
    - ready: Document successfully processed
    - failed: Document processing failed
    """
    doc_ref = db.collection("documents").document(document_id)
    doc = doc_ref.get()

    if not doc.exists:
        raise HTTPException(status_code=404, detail="Document not found")

    doc_data = doc.to_dict()
    return DocumentKGStatus(
        document_id=document_id,
        module_id=doc_data.get("module_id"),
        file_name=doc_data.get("file_name"),
        kg_status=KGStatus(doc_data.get("kg_status", "pending")),
        kg_processed_at=doc_data.get("kg_processed_at"),
        kg_error=doc_data.get("kg_error"),
        chunk_count=doc_data.get("kg_chunk_count"),
        entity_count=doc_data.get("kg_entity_count")
    )


@router.post("/process-batch", response_model=BatchProcessingResponse)
async def process_batch(request: BatchProcessingRequest):
    """
    Process multiple documents in batch for KG creation.

    - All documents must belong to the same module
    - Already processed documents are skipped (idempotent)
    - Returns task info for progress tracking
    """
    # Validate all documents exist and belong to module
    skipped_ids = []
    queued_ids = []

    for doc_id in request.file_ids:
        doc_ref = db.collection("documents").document(doc_id)
        doc = doc_ref.get()

        if not doc.exists:
            continue  # Skip non-existent docs

        doc_data = doc.to_dict()
        if doc_data.get("module_id") != request.module_id:
            raise HTTPException(
                status_code=400,
                detail=f"Document {doc_id} does not belong to module {request.module_id}"
            )

        # Check if already processed (idempotency)
        if doc_data.get("kg_status") == "ready":
            skipped_ids.append(doc_id)
        else:
            queued_ids.append(doc_id)

    if not queued_ids:
        return BatchProcessingResponse(
            task_id="",
            status_url="",
            documents_queued=0,
            documents_skipped=len(skipped_ids),
            message="All documents already processed"
        )

    # Trigger Celery task
    task = process_batch_task.delay(queued_ids, request.module_id, "staff_user")

    return BatchProcessingResponse(
        task_id=task.id,
        status_url=f"/api/v1/kg/tasks/{task.id}",
        documents_queued=len(queued_ids),
        documents_skipped=len(skipped_ids),
        message=f"Queued {len(queued_ids)} documents, skipped {len(skipped_ids)} already processed"
    )


@router.get("/processing-queue", response_model=List[ProcessingQueueItem])
async def get_processing_queue():
    """
    Get all documents currently being processed.

    Returns documents with kg_status == 'processing'.
    Used by frontend for ProcessingQueue component.
    """
    docs = db.collection("documents") \
        .where("kg_status", "==", "processing") \
        .stream()

    queue = []
    for doc in docs:
        doc_data = doc.to_dict()
        queue.append(ProcessingQueueItem(
            document_id=doc.id,
            module_id=doc_data.get("module_id"),
            file_name=doc_data.get("file_name"),
            status=KGStatus.PROCESSING,
            progress=doc_data.get("kg_progress", 0),
            step=doc_data.get("kg_step", "processing"),
            started_at=doc_data.get("kg_started_at", datetime.utcnow())
        ))

    return queue


@router.get("/tasks/{task_id}/status")
async def get_task_status(task_id: str):
    """
    Get Celery task status and progress.

    Returns task state and meta information.
    """
    task = process_batch_task.AsyncResult(task_id)
    return {
        "task_id": task_id,
        "status": task.state,
        "result": task.result if task.state == "SUCCESS" else None,
        "info": task.info if hasattr(task, "info") else None
    }
```
