```python
from celery import Celery
from celery.exceptions import SoftTimeLimitExceeded

app = Celery('aura_notes_tasks', broker='redis://localhost:6379/0')

@app.task(
    bind=True,
    autoretry_for=(ConnectionError, TimeoutError),
    retry_backoff=True,
    retry_backoff_max=600,
    retry_jitter=True,
    max_retries=5,
    acks_late=True,
    reject_on_worker_lost=True,
    time_limit=1800,          # 30 min hard limit
    soft_time_limit=1500      # 25 min soft limit
)
def process_document_task(self, document_id: str, module_id: str, user_id: str):
    """
    Process document into knowledge graph asynchronously.

    Progress tracking:
    - PENDING: Task received
    - PROCESSING: Document parsing
    - CHUNKING: Creating semantic chunks
    - EMBEDDING: Generating embeddings
    - EXTRACTING: Entity extraction
    - STORING: Neo4j storage
    - COMPLETED: Done
    - FAILED: Error
    """
    try:
        # Update initial state
        self.update_state(state='PROCESSING', meta={'progress': 5, 'step': 'Parsing document'})

        # Initialize processor
        processor = KnowledgeGraphProcessor(neo4j_driver, gemini_client)

        # Process document
        result = await processor.process_document(document_id, module_id, user_id)

        self.update_state(state='COMPLETED', meta={
            'progress': 100,
            'result': result
        })

        return result

    except SoftTimeLimitExceeded:
        self.update_state(state='FAILED', meta={'error': 'Task timed out'})
        raise
    except Exception as e:
        self.update_state(state='FAILED', meta={'error': str(e)})
        raise


@app.task(
    bind=True,
    autoretry_for=(ConnectionError, TimeoutError),
    retry_backoff=True,
    max_retries=3,
    acks_late=True
)
def process_batch_task(self, document_ids: List[str], module_id: str, user_id: str):
    """Process multiple documents in batch."""
    results = []
    total = len(document_ids)

    for i, doc_id in enumerate(document_ids):
        self.update_state(state='PROCESSING', meta={
            'progress': (i / total) * 100,
            'current': i + 1,
            'total': total
        })

        result = process_document_task.delay(doc_id, module_id, user_id)
        results.append({'document_id': doc_id, 'task_id': result.id})

    return results
```
