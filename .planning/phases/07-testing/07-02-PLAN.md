  http.get('/api/sessions', ({ request }) => {

    return HttpResponse.json({
      sessions: [
        { id: '1', title: 'Study Session', created_at: new Date().toISOString() },
      ],
    });
  }),

  // Create session
  http.post('/api/sessions', async ({ request }) => {
    const body = await request.json();
    return HttpResponse.json({
      id: `session-${Date.now()}`,
      title: body.title,
      created_at: new Date().toISOString(),
    });
  }),

  // Delete session
  http.delete('/api/sessions/:sessionId', ({ params }) => {
    return HttpResponse.json({ success: true });
  }),

  // Session messages
  http.get('/api/sessions/:sessionId/messages', ({ params }) => {
    return HttpResponse.json({
      messages: [
        {
          id: '1',
          role: 'user',
          content: 'Hello',
          created_at: new Date().toISOString(),
        },
      ],
    });
  }),

  // Session query (chat)
  http.post('/api/sessions/:sessionId/query', async ({ request }) => {
    const body = await request.json();
    return HttpResponse.json({
      id: `msg-${Date.now()}`,
      role: 'assistant',
      content: `You asked: ${body.query}`,
      sources: [
        {
          doc_id: '1',
          title: 'Test Document',
          text: 'Relevant text',
          score: 0.85,
        },
      ],
    });
  }),
];
