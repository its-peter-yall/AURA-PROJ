```python
from fastapi import Request

@app.middleware("http")
async def metrics_middleware(request: Request, call_next):
    method = request.method
    endpoint = request.url.path

    start_time = time.time()
    try:
        response = await call_next(request)
        status_code = response.status_code
    except Exception as e:
        status_code = 500
        raise
    finally:
        duration = time.time() - start_time
        HTTP_REQUESTS_TOTAL.labels(
            method=method,
            endpoint=endpoint,
            status_code=status_code
        ).inc()
        HTTP_REQUEST_DURATION.labels(
            method=method,
            endpoint=endpoint
        ).observe(duration)

    return response
```

```python
def _check_dependencies() -> tuple[bool, bool, bool, Dict[str, float]]:
    """Check Neo4j and Redis connectivity with latency tracking."""
    neo4j_latency = None
    redis_latency = None

    try:
        start = time.time()
        gm = get_graph_manager()
        if gm and gm.driver:
            gm.driver.verify_connectivity()
            neo4j_latency = (time.time() - start) * 1000
            neo4j_connected = True
    except Exception as e:
        neo4j_connected = False

    try:
        start = time.time()
        if config.REDIS_ENABLED:
            redis_client = get_redis()
            if redis_client:
                redis_latency = (time.time() - start) * 1000
                redis_connected = True
        else:
            redis_connected = True
    except Exception:
        redis_connected = False

    services_ready = neo4j_connected and redis_connected
    return neo4j_connected, redis_connected, services_ready, {
        "neo4j": neo4j_latency,
        "redis": redis_latency
```

```python
from starlette.middleware.base import BaseHTTPMiddleware

MAX_REQUEST_SIZE_MB = 200

class RequestSizeMiddleware(BaseHTTPMiddleware):
    """Limit request body size."""

    async def dispatch(self, request: Request, call_next: Callable) -> Response:
        content_length = request.headers.get("content-length")
        if content_length:
            max_size = MAX_REQUEST_SIZE_MB * 1024 * 1024
            if int(content_length) > max_size:
                return JSONResponse(
                    status_code=413,
                    content={
                        "success": False,
                        "error_code": "REQUEST_TOO_LARGE",
                        "message": f"Request body exceeds {MAX_REQUEST_SIZE_MB}MB limit"
```

```python
from locust import HttpUser, task, between, events
from locust.runners import Runner

class ChatUser(HttpUser):
    wait_time = between(1, 5)  # Wait 1-5 seconds between tasks

    @task(3)
    def send_query(self):
        """Send RAG query (highest weight)."""
        self.client.post(
            f"/chat/sessions/{self.session_id}/messages",
            json={"content": "What are the key concepts in module X?"},
            headers={"Authorization": f"Bearer {self.token}"}
        )

    @task(2)
    def list_sessions(self):
        """List user's chat sessions."""
        self.client.get("/chat/sessions", headers={"Authorization": f"Bearer {self.token}"})

    @task(1)
    def get_messages(self):
        """Get message history for a session."""
        self.client.get(
            f"/chat/sessions/{self.session_id}/messages",
            headers={"Authorization": f"Bearer {self.token}"}
        )

    @task(1)
    def list_modules(self):
        """List available modules."""
        self.client.get("/modules", headers={"Authorization": f"Bearer {self.token}"})

    def on_start(self):
        """Setup user session."""
        self.session_id = self.create_session()
        self.token = self.login()

class ModuleUser(HttpUser):
    wait_time = between(2, 10)
    weight = 0.2

    @task(3)
    def list_modules(self):
        """List modules."""
        self.client.get("/modules")

    @task(2)
    def get_module_details(self):
        """Get module details."""
        self.client.get(f"/modules/{self.module_id}")

    @task(1)
    def create_module(self):
        """Create new module."""
        self.client.post("/modules", json={"name": "Test Module"})

class DocumentUser(HttpUser):
    wait_time = between(5, 30)
    weight = 0.2

    @task(3)
    def list_documents(self):
        """List documents."""
        self.client.get("/documents")

    @task(2)
    def get_document(self):
        """Get document details."""
        self.client.get(f"/documents/{self.document_id}")

    @task(1)
    def upload_document(self):
        """Upload document (heavy operation)."""
        with open("test_document.pdf", "rb") as f:
            self.client.post(
                "/documents/upload",
                files={"file": ("test_document.pdf", f, "application/pdf")},
                headers={"Authorization": f"Bearer {self.token}"}
            )
```

```python
# tests/security/test_middleware.py
"""Security middleware tests for CORS, rate limiting, and security headers."""

import pytest
from fastapi.testclient import TestClient


def test_cors_headers_present(client: TestClient):
    """Verify CORS headers on all responses."""
    response = client.get("/health")
    assert response.headers.get("Access-Control-Allow-Origin") is not None
    assert response.headers.get("Access-Control-Allow-Credentials") == "true"
    assert response.headers.get("Access-Control-Allow-Methods") is not None
    assert response.headers.get("Access-Control-Allow-Headers") is not None


def test_security_headers_present(client: TestClient):
    """Verify security headers on all responses."""
    response = client.get("/health")
    assert response.headers["X-Content-Type-Options"] == "nosniff"
    assert response.headers["X-Frame-Options"] == "DENY"
    assert response.headers["X-XSS-Protection"] == "1; mode=block"
    assert "strict-origin" in response.headers["Referrer-Policy"]
    assert "geolocation=()" in response.headers["Permissions-Policy"]


def test_rate_limiting_enforced(client: TestClient):
    """Verify rate limiting is active after limit exceeded."""
    # Make requests up to the limit (100/minute by default)
    for _ in range(100):
        response = client.get("/health")
        assert response.status_code == 200

    # 101st request should be rate limited
    response = client.get("/health")
    assert response.status_code == 429


def test_rate_limit_headers(client: TestClient):
    """Verify rate limit headers are present."""
    response = client.get("/health")
    assert "X-RateLimit-Limit" in response.headers
    assert "X-RateLimit-Remaining" in response.headers
    assert "X-RateLimit-Reset" in response.headers


def test_rate_limit_headers_on_429(client: TestClient):
    """Verify rate limit headers on 429 response."""
    # Exceed rate limit
    for _ in range(101):
        client.get("/health")

    response = client.get("/health")
    assert response.status_code == 429
    assert "X-RateLimit-Limit" in response.headers
    assert "X-RateLimit-Remaining" in response.headers
    assert "X-RateLimit-Reset" in response.headers
    assert "Retry-After" in response.headers


def test_hsts_not_on_http(client: TestClient):
    """HSTS header should not be present on HTTP requests."""
    response = client.get("/health")
    assert "Strict-Transport-Security" not in response.headers


def test_request_id_generated(client: TestClient):
    """Verify unique request ID on each request."""
    response1 = client.get("/health")
    response2 = client.get("/health")

    assert "X-Request-ID" in response1.headers
    assert "X-Request-ID" in response2.headers
    assert response1.headers["X-Request-ID"] != response2.headers["X-Request-ID"]
```
