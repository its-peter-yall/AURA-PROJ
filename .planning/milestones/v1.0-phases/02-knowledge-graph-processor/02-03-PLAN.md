# Entity extraction prompt
ENTITY_EXTRACTION_PROMPT = """
You are a knowledge graph expert. Extract entities from the following text.

Text:
{chunk_text}

Extract entities of these types:
- TOPIC: Main subject areas (e.g., "Machine Learning", "Database Systems")
- CONCEPT: Specific ideas or definitions (e.g., "Neural Network", "SQL Injection")
- METHODOLOGY: Approaches or techniques (e.g., "Backpropagation", "CRUD Operations")
- FINDING: Important discoveries or conclusions

Return JSON array:
[
  {{
    "name": "Entity Name",
    "category": "TOPIC|CONCEPT|METHODOLOGY|FINDING",
    "definition": "Brief definition (1-2 sentences)",
    "context_snippet": "Source text excerpt",
    "confidence": 0.95
  }}
]

Only extract entities that are explicitly defined or clearly explained.
"""

async def extract_entities(self, chunk: Dict, module_id: str) -> List[Entity]:
    """Extract entities from a chunk using Gemini."""
    prompt = ENTITY_EXTRACTION_PROMPT.format(chunk_text=chunk['content'][:5000])
    response = await self.gemini.generate(prompt)

    entities = json.loads(response)

    # Add module_id to each entity
    for entity in entities:
        entity['module_id'] = module_id
        entity['chunk_id'] = chunk.get('id')

    return entities

async def _store_entities(self, entities: List[Entity], chunk_id: str, module_id: str):
    """Store entities in Neo4j with relationships."""
    query = """
    UNWIND $entities AS entity
    MERGE (e:Entity {id: apoc.create.uuid()})
    SET e.name = entity.name,
        e.category = entity.category,
        e.definition = entity.definition,
        e.context_snippet = entity.context_snippet,
        e.confidence = entity.confidence,
        e.module_id = entity.module_id,
        e.chunk_id = entity.chunk_id
    WITH e, chunk_id
    MATCH (c:Chunk {id: chunk_id})
    MERGE (c)-[:CONTAINS_ENTITY {relevance_score: entity.confidence}]->(e)
    """
