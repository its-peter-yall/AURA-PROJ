MATCH (e:Entity)
WHERE e.module_id IN $module_ids

MATCH (e2:Entity)
WHERE e2.name = e.name
AND e2.module_id IN $module_ids
AND e2.module_id <> e.module_id

WITH e.name as concept_name,
     collect(DISTINCT e.module_id) as modules,
     e.definition as definition

WHERE size(modules) >= 2

RETURN {
    name: concept_name,
    definition: definition,
    appears_in_modules: modules
} as cross_concept
ORDER BY size(modules) DESC
LIMIT $top_k

```python
async def find_cross_module_concepts(
    self,
    query: str,
    module_ids: List[str],
    top_k: int = 5
) -> List[Dict[str, Any]]:
    """
    Find concepts appearing in 2+ selected modules.

    Args:
        query: User query for context
        module_ids: List of module IDs to analyze
        top_k: Maximum concepts to return

    Returns:
        [{"name": str, "definition": str, "modules": [str]}]
    """
    try:
        # Get entities from all selected modules
        entities = self.graph_manager.get_module_entities(module_ids)

        # Find concepts that appear in multiple modules
        concept_modules: Dict[str, Dict[str, Any]] = {}

        for entity in entities:
            name = entity.get("name")
            module_id = entity.get("module_id")

            if not name or not module_id:
                continue

            if name not in concept_modules:
                concept_modules[name] = {
                    "name": name,
                    "definition": entity.get("definition", ""),
                    "modules": []
                }

            if module_id not in concept_modules[name]["modules"]:
                concept_modules[name]["modules"].append(module_id)

        # Filter to concepts in 2+ modules
        cross_concepts = [
            c for c in concept_modules.values()
            if len(c["modules"]) >= 2
        ]

        # Sort by number of modules and limit
        cross_concepts.sort(key=lambda x: len(x["modules"]), reverse=True)
        return cross_concepts[:top_k]

    except Exception as e:
        logger.warning(f"Cross-module concept discovery failed: {e}")
        return []
```
