def get_module_documents(self, module_ids: List[str]) -> List[Dict[str, Any]]:
    """
    Get all documents belonging to the specified module(s).

    Args:
        module_ids: List of module IDs to filter by

    Returns:
        List of document dictionaries with id, title, authors, etc.
    """
    if not module_ids:
        return []

    query = """
    MATCH (d:Document)-[:BELONGS_TO_MODULE]->(m:Module)
    WHERE m.id IN $module_ids
    RETURN d
    ORDER BY d.upload_date DESC
    """
    try:
        results = self.execute_query(query, {"module_ids": module_ids})
        return [record['d'] for record in results if 'd' in record]
    except Exception as e:
        logger.error(f"Failed to get module documents: {e}")
        return []

def search_vector_with_modules(
    self,
    query_embedding: List[float],
    module_ids: List[str],
    top_k: int = 10
) -> List[Dict[str, Any]]:
    """
    Perform vector similarity search restricted to specific modules.

    Args:
        query_embedding: The embedding vector for similarity search
        module_ids: List of module IDs to filter by
        top_k: Maximum number of results

    Returns:
        List of chunks with scores and document metadata
    """
    if not module_ids:
        return []

    try:
        # Use Neo4j vector index with module filter
        query = """
        CALL db.index.vector.queryNodes('chunk_vector_index', $top_k, $query_embedding)
        YIELD node as c, score
        MATCH (d:Document)-[:HAS_CHUNK]->(c)
        WHERE d.module_id IN $module_ids
        RETURN d.id as doc_id, d.title as title, d.authors as authors,
               c.text as text, c.index as chunk_index, c.id as chunk_id, score
        ORDER BY score DESC
        """
        results = self.execute_query(query, {
            "query_embedding": query_embedding,
            "top_k": top_k * 2,  # Get more to filter
            "module_ids": module_ids
        })

        if results:
            return results[:top_k]

    except Exception as e:
        logger.warning(f"Neo4j vector search with modules failed: {e}")

    # Fallback: Python cosine similarity with module filter
    return self._search_chunks_with_modules_fallback(
        query_embedding, module_ids, top_k
    )

def get_cross_module_concepts(
    self,
    module_ids: List[str],
    top_k: int = 5
) -> List[Dict[str, Any]]:
    """
    Find entities/concepts that appear in 2+ selected modules.

    Args:
        module_ids: List of module IDs to analyze
        top_k: Maximum number of concepts to return

    Returns:
        [{"name": str, "definition": str, "modules": [str], "module_count": int}]
    """
    if not module_ids or len(module_ids) < 2:
        return []

    try:
        # Find concepts that appear in multiple modules
        query = """
        // Get all entities from selected modules
        MATCH (e:Entity)
        WHERE e.module_id IN $module_ids

        // Group by entity name
        WITH e.name as concept_name, collect(DISTINCT e.module_id) as modules

        // Filter to concepts in 2+ modules
        WHERE size(modules) >= 2

        // Get definition from one instance
        MATCH (e2:Entity {name: concept_name})
        WHERE e2.module_id IN $module_ids
        WITH concept_name, modules, e2.definition as definition
        LIMIT 1

        RETURN {
            name: concept_name,
            definition: definition,
            appears_in_modules: modules,
            module_count: size(modules)
        } as cross_concept
        ORDER BY module_count DESC
        LIMIT $top_k
        """
        results = self.execute_query(query, {
            "module_ids": module_ids,
            "top_k": top_k
        })

        return [r['cross_concept'] for r in results if 'cross_concept' in r]

    except Exception as e:
        logger.error(f"Failed to get cross-module concepts: {e}")
        return []

def get_module_entities(
    self,
    module_ids: List[str],
    entity_types: Optional[List[str]] = None
) -> List[Dict[str, Any]]:
    """
    Get all entities belonging to the specified module(s).

    Args:
        module_ids: List of module IDs to filter by
        entity_types: Optional list of entity types (Topic, Concept, etc.)

    Returns:
        List of entity dictionaries with id, name, type, module_id
    """
    if not module_ids:
        return []

    try:
        # Build entity type filter
        if entity_types:
            type_where = "AND labels(e)[0] IN $entity_types"
            params = {"module_ids": module_ids, "entity_types": entity_types}
        else:
            type_where = ""
            params = {"module_ids": module_ids}

        query = f"""
        MATCH (e:Entity)
        WHERE e.module_id IN $module_ids {type_where}

        RETURN e, labels(e)[0] as type, e.module_id as module_id
        ORDER BY e.name
        """

        results = self.execute_query(query, params)

        return [
            {
                **dict(record['e']),
                "type": record['type'],
                "module_id": record['module_id']
            }
            for record in results
        ]

    except Exception as e:
        logger.error(f"Failed to get module entities: {e}")
        return []
```

def get_document_module_id(self, doc_id: str) -> Optional[str]:
    """
    Get the module ID that a document belongs to.

    Args:
        doc_id: The document ID

    Returns:
        Module ID or None if document not in any module
    """
    try:
        query = """
        MATCH (d:Document {id: $doc_id})-[:BELONGS_TO_MODULE]->(m:Module)
        RETURN m.id as module_id
        """
        result = self.execute_query(query, {"doc_id": doc_id})
        return result[0]['module_id'] if result else None
    except Exception as e:
        logger.error(f"Failed to get document module ID: {e}")
        return None

def get_module_chunks(
    self,
    module_ids: List[str],
    top_k: int = 50
) -> List[Dict[str, Any]]:
    """
    Get chunks from documents in the specified module(s).
    Used for vector search with module filtering.

    Args:
        module_ids: List of module IDs to filter by
        top_k: Maximum number of chunks to return

    Returns:
        List of chunk dictionaries with text, index, doc_id
    """
    if not module_ids:
        return []

    try:
        query = """
        MATCH (d:Document)-[:HAS_CHUNK]->(c:Chunk)
        WHERE d.module_id IN $module_ids AND c.embedding IS NOT NULL
        RETURN d.id as doc_id, d.title as title, d.authors as authors,
               c.text as text, c.index as chunk_index, c.id as chunk_id,
               c.embedding as embedding
        ORDER BY c.index
        LIMIT $top_k
        """
        results = self.execute_query(query, {
            "module_ids": module_ids,
            "top_k": top_k
        })
        return results

    except Exception as e:
        logger.error(f"Failed to get module chunks: {e}")
        return []
```
