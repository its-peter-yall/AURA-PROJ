**router.py — Add three public metadata methods to ModelRouter class:**

1. `list_models(provider=None)` — aggregates or filters models:
```python
async def list_models(
    self,
    provider: ProviderType | str | None = None,
) -> list[ModelInfo]:
    """List available models from registered providers."""
    if provider is not None:
        provider_type = _coerce_provider_type(provider)
        resolved = self._providers.get(provider_type)
        if resolved is None:
            raise ModelUnavailableError(
                f'No provider registered for {provider_type.value}',
                provider=provider_type.value,
            )
        return await resolved.list_models()

    all_models: list[ModelInfo] = []
    for registered_provider in self._providers.values():
        all_models.extend(await registered_provider.list_models())
    return all_models
```

2. `health_check(provider=None)` — checks one or all providers:
```python
async def health_check(
    self,
    provider: ProviderType | str | None = None,
) -> dict[ProviderType, bool]:
    """Check health of one or all registered providers."""
    if provider is not None:
        provider_type = _coerce_provider_type(provider)
        resolved = self._providers.get(provider_type)
        if resolved is None:
            return {provider_type: False}
        return {provider_type: await resolved.health_check()}

    results: dict[ProviderType, bool] = {}
    for provider_type, registered_provider in self._providers.items():
        results[provider_type] = await registered_provider.health_check()
    return results
```

3. `get_provider(provider_type)` — returns raw provider instance for provider-specific operations like `get_credit_balance()`:
```python
def get_provider(
    self,
    provider_type: ProviderType | str,
) -> BaseProvider:
    """Return a registered provider for direct access to provider-specific APIs."""
    resolved_type = _coerce_provider_type(provider_type)
    resolved = self._providers.get(resolved_type)
    if resolved is None:
        raise ModelUnavailableError(
            f'No provider registered for {resolved_type.value}',
            provider=resolved_type.value,
        )
    return resolved
```
