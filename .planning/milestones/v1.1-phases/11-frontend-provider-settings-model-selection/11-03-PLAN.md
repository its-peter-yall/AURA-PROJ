    ```typescript
    export function formatModelLabel(modelId: string, displayName?: string | null): string {
        // Use API-provided display_name if available
        if (displayName) return displayName;

        // Hard-coded fallback for known Gemini models
        const modelMap: Record<string, string> = {
            'gemini-2.0-flash-001': 'Gemini 2 Flash',
            'gemini-2.5-flash-lite': 'Gemini 2.5 Flash Lite',
            'gemini-2.5-flash': 'Gemini 2.5 Flash',
            'gemini-3-flash-preview': 'Gemini 3 Flash',
        };
        if (modelMap[modelId]) return modelMap[modelId];

        // OpenRouter format: 'vendor/model-name' → extract model part
        const slashIndex = modelId.indexOf('/');
        const nameToFormat = slashIndex >= 0 ? modelId.substring(slashIndex + 1) : modelId;

        // Convert identifier to title case
        return nameToFormat
            .replace(/-/g, ' ')
            .split(' ')
            .map(word => word.charAt(0).toUpperCase() + word.slice(1).toLowerCase())
            .join(' ');
    }
    ```

    This preserves backward compatibility (existing callers pass only modelId) while enabling InlineModelPicker to pass display_name from the API. The OpenRouter handling strips the `vendor/` prefix before formatting.
