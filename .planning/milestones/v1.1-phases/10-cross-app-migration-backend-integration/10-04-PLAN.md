    """Return a model_router-backed model for genai call sites."""
    return VertexCompatModel(model_name)

def generate_content_with_thinking(model, prompt: str):
    """Generate content with thinking support via model_router."""
    if hasattr(model, 'generate_content'):
        return model.generate_content(

    )
