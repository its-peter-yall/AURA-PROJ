### 2. Create verify_phase_1.py script

**File:** `AURA-NOTES-MANAGER/api/verify_phase_1.py`

```python
#!/usr/bin/env python3
"""
verify_phase_1.py
Verification script for Phase 1: Configuration & Client Unification
"""

import sys
from pathlib import Path

project_root = Path(__file__).parent.parent
sys.path.insert(0, str(project_root))

import os
os.chdir(project_root)

def check_no_google_generativeai():
    """Check no direct google.generativeai imports remain."""
    import subprocess
    result = subprocess.run(
        ["grep", "-rn", "import google.generativeai", "api/", "services/"],
        capture_output=True, text=True
    )
    count = len(result.stdout.strip().split("\n")) if result.stdout.strip() else 0
    if count == 0:
        print("✅ PASS: No google.generativeai imports")
        return True
    else:
        print(f"❌ FAIL: Found {count} google.generativeai imports")
        print(result.stdout)
        return False

def check_vertex_ai_client_usage():
    """Check vertex_ai_client is being used."""
    import subprocess
    result = subprocess.run(
        ["grep", "-rn", "from services.vertex_ai_client", "api/", "services/"],
        capture_output=True, text=True
    )
    count = len(result.stdout.strip().split("\n")) if result.stdout.strip() else 0
    if count > 0:
        print(f"✅ PASS: Found {count} vertex_ai_client imports")
        return True
    else:
        print("❌ FAIL: No vertex_ai_client usage found")
        return False

def check_config_imports():
    """Check config imports work."""
    try:
        from api.config import LLM_ENTITY_EXTRACTION_MODEL, EMBEDDING_MODEL
        print(f"✅ PASS: Config imports work - Extraction={LLM_ENTITY_EXTRACTION_MODEL}, Embedding={EMBEDDING_MODEL}")
        return True
    except Exception as e:
        print(f"❌ FAIL: Config import error - {e}")
        return False

def check_kg_processor_config():
    """Check kg_processor uses config."""
    try:
        from api.kg_processor import GeminiClient
        client = GeminiClient()
        print(f"✅ PASS: kg_processor uses config - Extraction={client.extraction_model}, Embedding={client.embedding_model}")
        return True
    except Exception as e:
        print(f"❌ FAIL: kg_processor config error - {e}")
        return False

def check_embeddings_config():
    """Check embeddings.py uses config."""
    try:
        from services.embeddings import EMBEDDING_MODEL
        print(f"✅ PASS: embeddings.py uses config - Embedding={EMBEDDING_MODEL}")
        return True
    except Exception as e:
        print(f"❌ FAIL: embeddings.py config error - {e}")
        return False

def check_no_api_key():
    """Check no GEMINI_API_KEY usage."""
    import subprocess
    result = subprocess.run(
        ["grep", "-rn", "GEMINI_API_KEY", "api/", "services/"],
        capture_output=True, text=True
    )
    count = len(result.stdout.strip().split("\n")) if result.stdout.strip() else 0
    if count == 0:
        print("✅ PASS: No GEMINI_API_KEY references")
        return True
    else:
        print(f"❌ FAIL: Found {count} GEMINI_API_KEY references")
        print(result.stdout)
        return False

def main():
    print("=" * 50)
    print("Phase 1 Verification: Configuration & Client Unification")
    print("=" * 50)
    print("")

    checks = [
        ("No google.generativeai imports", check_no_google_generativeai),
        ("vertex_ai_client usage", check_vertex_ai_client_usage),
        ("Config imports", check_config_imports),
        ("kg_processor uses config", check_kg_processor_config),
        ("embeddings.py uses config", check_embeddings_config),
        ("No GEMINI_API_KEY usage", check_no_api_key),
    ]

    results = []
    for name, check_func in checks:
        print(f"Checking: {name}...")
        results.append(check_func())
        print("")

    print("=" * 50)
    print("Summary")
    print("=" * 50)
    passed = sum(results)
    total = len(results)
    print(f"Passed: {passed}/{total}")

    if all(results):
        print("✅ Phase 1 VERIFICATION PASSED")
        return 0
    else:
        print("❌ Phase 1 VERIFICATION FAILED")
        return 1

if __name__ == "__main__":
    sys.exit(main())
```
