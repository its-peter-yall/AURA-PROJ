```python
async def process_document(self, file_path, module_id, user_id, document_id):
    # Step 1: Extract text from file
    text = self.extract_text_from_file(file_path)
    
    # Step 2: Use entity_aware_chunker
    chunks = self.chunker.chunk_text_hierarchical(text, document_id)
    
    # Step 3: Generate embeddings
    chunk_texts = [c.text for c in chunks]
    embeddings = await self._generate_chunk_embeddings(chunk_texts)
    
    # Step 4: Use llm_entity_extractor
    extraction_results = await self._extract_entities_from_chunks(chunk_texts)
    entities = merge_extraction_results(extraction_results)
    
    # Step 5: Store in graph
    result = self.store_in_graph(document_id, file_path, chunks, embeddings, entities, module_id, user_id)
    
    return result
```

### 5. Add helper methods

**Add these methods to KnowledgeGraphProcessor:**

```python
async def _generate_chunk_embeddings(self, chunk_texts):
    """Generate embeddings using services.embeddings.get_embedding"""
    embeddings = []
    for text in chunk_texts:
        embedding = await get_embedding(text)
        embeddings.append(embedding)
    return embeddings

async def _extract_entities_from_chunks(self, chunk_texts):
    """Extract entities using services.llm_entity_extractor.extract_entities"""
    results = []
    for i, text in enumerate(chunk_texts):
        result = await extract_entities(text, f"chunk_{i}")
        results.append(result)
    return results
```
