```python
    retry,
    stop_after_attempt,
    wait_exponential,
    retry_if_exception_type,
    before_sleep_log,
    after_log
)
import logging

logger = logging.getLogger(__name__)
```

### 3. Add custom exceptions

```python
class ExtractionError(Exception): pass
class VertexAIError(Exception): pass
class EmbeddingError(Exception): pass
```

### 4. Add retry decorator for entity extraction

```python
@retry(
    stop=stop_after_attempt(3),
    wait=wait_exponential(multiplier=1, min=2, max=60),
    retry=retry_if_exception_type((VertexAIError, ExtractionError)),
    before_sleep=before_sleep_log(logger, logging.WARNING)
)
async def extract_entities_with_retry(text, chunk_id):
    """Extract entities with automatic retry on failure."""
    from services.llm_entity_extractor import extract_entities
    return await extract_entities(text, chunk_id)
```

### 5. Add retry decorator for embeddings

```python
@retry(
    stop=stop_after_attempt(3),
    wait=wait_exponential(multiplier=1, min=1, max=30),
    retry=retry_if_exception_type((VertexAIError, EmbeddingError)),
    before_sleep=before_sleep_log(logger, logging.WARNING)
)
async def generate_embedding_with_retry(text):
    """Generate embedding with automatic retry on failure."""
    from services.embeddings import get_embedding
    return await get_embedding(text)
```

### 6. Update processor to use retry helpers

**In KnowledgeGraphProcessor:**

```python
async def _generate_chunk_embeddings(self, chunk_texts):
    embeddings = []
    for i, text in enumerate(chunk_texts):
        try:
            embedding = await generate_embedding_with_retry(text)
            embeddings.append(embedding)
        except Exception as e:
            logger.error(f"Embedding failed for chunk {i}: {e}")
            embeddings.append([0.0] * 768)  # Placeholder
    return embeddings

async def _extract_entities_from_chunks(self, chunk_texts):
    tasks = [extract_entities_with_retry(text, f"chunk_{i}") for i, text in enumerate(chunk_texts)]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    return [r for r in results if not isinstance(r, Exception)]
```
