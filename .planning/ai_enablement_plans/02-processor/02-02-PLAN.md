import asyncio
from services.llm_entity_extractor import extract_entities, merge_extraction_results

    Neural networks are computing systems inspired by biological neural networks.
    Researchers at Google have contributed significantly.'''

    result = await extract_entities(sample_text, 'test-chunk')
    
    print(f'Extracted {len(result.entities)} entities:')

    print(f'Extracted {len(result.relationships)} relationships')

asyncio.run(test())
